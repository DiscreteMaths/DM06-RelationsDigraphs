\frametitle{Cartesian Product}
{\LARGE
\begin{itemize}
\item Let $X$ and $Y$ be sets.
\item The \textbf{cartesian product} $X \times Y$ is the set whose elements are \textbf{all} of the ordered pairs of elements $(x,y)$ where $x \in X$ and $y \in Y$.
\end{itemize}
